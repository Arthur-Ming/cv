function repeater(str, {
   repeatTimes = 1,
   separator = '+',
   addition = "",
   additionRepeatTimes = 1,
   additionSeparator = '|'
}) {

   const result = Array(repeatTimes)
      .fill(`${str}${addition !== '' ? 
         repeater(addition, {
         repeatTimes: additionRepeatTimes,
         separator: additionSeparator,
      }) : ''}`)

   return result.join(separator)
}
